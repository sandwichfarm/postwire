Write `src/transport/seq.ts` with the EXACT algorithm from 01-RESEARCH.md "Sequence Number
Arithmetic (PROTO-02, SESS-06)":
```typescript
const SEQ_BITS: number = 32
const SEQ_MASK: number = 0xffff_ffff
const HALF_WINDOW: number = 0x8000_0000  // 2^31

export function seqMask(n: number): number { return n >>> 0 }
export function seqLT(a: number, b: number): boolean {
  return ((seqMask(a) - seqMask(b)) >>> 0) > HALF_WINDOW
}
export function seqGT(a: number, b: number): boolean { return seqLT(b, a) }
export function seqLTE(a: number, b: number): boolean { return !seqGT(a, b) }
export function seqNext(n: number): number { return (n + 1) >>> 0 }
export { SEQ_BITS, SEQ_MASK, HALF_WINDOW }
```

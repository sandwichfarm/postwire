  SESS-01 behaviors (unit tests — describe 'ReorderBuffer'):
  - in-order delivery: insert frame with seqNum === initSeq → returns [frame], nextExpected advances
  - consecutive flush: insert seq=5 then seq=3 when nextExpected=3 → second insert flushes [seq=3, seq=4? NO — inserts seq=3 first (buffers), then seq=5 (buffers), then seq=3 arrives when nextExpected=3 → flush seq=3 only; seq=5 remains]. Revise: insert seq=3 at nextExpected=3 → returns [seq=3]. Insert seq=5 into buffer. Insert seq=4 when nextExpected=4 → returns [seq=4, seq=5] (drains the map).
  - out-of-order buffering: insert seq=nextExpected+2 → returns [], buffer.size===1
  - overflow: insert maxReorderBuffer+1 distinct out-of-order frames → throws Error('REORDER_OVERFLOW')
  - duplicate drop: insert seq=0 when nextExpected=5 (seqLT(0, 5) is true in linear space) → returns [] silently
  - duplicate key in buffer: insert same seqNum twice while buffered → second is silently dropped (no-op return [])
  - SESS-06 wraparound fuzz (fast-check): starting at initSeq=0xFFFFFFF0, insert 32 frames in shuffled order (fc.shuffledSubarray of seq array from 0xFFFFFFF0 to 0x0000000F), assert all 32 are delivered exactly once in correct seq order
  - wraparound deterministic: starting at initSeq=0xFFFFFFF0, insert frames 0x0 through 0xF first (buffered), then 0xFFFFFFF0 through 0xFFFFFFFF in order → all 32 delivered in correct order

  ```typescript
  // src/session/reorder-buffer.ts
  import { seqLT, seqNext } from '../transport/seq.js';
  import type { DataFrame } from '../framing/types.js';

  export interface ReorderBufferOptions {
    maxReorderBuffer?: number;  // default: 64
  }

  export class ReorderBuffer {
    readonly #buffer = new Map<number, DataFrame>();
    #nextExpected: number;
    readonly #maxBuffer: number;

    constructor(initSeq: number, opts: ReorderBufferOptions = {}) {
      this.#nextExpected = initSeq >>> 0;
      this.#maxBuffer = opts.maxReorderBuffer ?? 64;
    }

    get nextExpected(): number { return this.#nextExpected; }

    insert(frame: DataFrame): DataFrame[] {
      const seq = frame.seqNum >>> 0;

      // Already delivered (seqLT handles wraparound correctly):
      if (seqLT(seq, this.#nextExpected)) return [];

      // Exact duplicate already in buffer:
      if (this.#buffer.has(seq) && seq !== this.#nextExpected) return [];

      // In-order frame — deliver and drain consecutive:
      if (seq === this.#nextExpected) {
        const out: DataFrame[] = [frame];
        this.#nextExpected = seqNext(this.#nextExpected);
        while (this.#buffer.has(this.#nextExpected)) {
          out.push(this.#buffer.get(this.#nextExpected)!);
          this.#buffer.delete(this.#nextExpected);
          this.#nextExpected = seqNext(this.#nextExpected);
        }
        return out;
      }

      // Out-of-order: capacity check before buffering:
      if (this.#buffer.size >= this.#maxBuffer) {
        throw new Error('REORDER_OVERFLOW');
      }
      this.#buffer.set(seq, frame);
      return [];
    }
  }
  ```

  - NEVER use `seq < this.#nextExpected` (raw comparison) — always `seqLT(seq, this.#nextExpected)`
  - seqNext() for incrementing, never `+1` directly

  it('delivers all 32 frames in order through 0xFFFFFFF0 wrap (SESS-06)', () => {
    const START = 0xFFFFFFF0;
    const COUNT = 32;
    const seqs: number[] = [];
    for (let i = 0; i < COUNT; i++) seqs.push((START + i) >>> 0);
    
    fc.assert(
      fc.property(fc.shuffledSubarray(seqs, { minLength: COUNT, maxLength: COUNT }), (shuffled) => {
        const buf = new ReorderBuffer(START);
        const delivered: number[] = [];
        for (const seq of shuffled) {
          delivered.push(...buf.insert(makeFrame(seq)).map(f => f.seqNum));
        }
        expect(delivered).toEqual(seqs);
      }),

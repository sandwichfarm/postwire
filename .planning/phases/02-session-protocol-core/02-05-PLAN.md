  ```typescript
  // src/session/index.ts
  import { FRAME_MARKER } from '../framing/types.js';
  import type { Frame, DataFrame, ChunkType } from '../framing/types.js';
  import { ReorderBuffer } from './reorder-buffer.js';
  import type { ReorderBufferOptions } from './reorder-buffer.js';
  import { CreditWindow } from './credit-window.js';
  import type { CreditWindowOptions } from './credit-window.js';
  import { Chunker } from './chunker.js';
  import type { ChunkerOptions, ChunkResult } from './chunker.js';
  import { transition, isTerminalState } from './fsm.js';
  import type { StreamState, StreamEvent } from './fsm.js';

  export type { StreamState, StreamEvent, IllegalTransitionError } from './fsm.js';
  export type { ReorderBufferOptions } from './reorder-buffer.js';
  export type { CreditWindowOptions } from './credit-window.js';
  export type { ChunkerOptions, ChunkResult } from './chunker.js';

  export interface SessionOptions {
    channelId: string;
    streamId: number;
    role: 'initiator' | 'responder';
    maxReorderBuffer?: number;
    initialCredit?: number;
    highWaterMark?: number;
    maxChunkSize?: number;
    stallTimeoutMs?: number;
    onMetrics?: (event: never) => void;  // Phase 4 extension point — never in Phase 2
  }

  export class Session {
    readonly #channelId: string;
    readonly #streamId: number;
    readonly #role: 'initiator' | 'responder';
    readonly #initialCredit: number;
    #state: StreamState = 'IDLE';

    readonly #reorder: ReorderBuffer;
    readonly #credit: CreditWindow;
    readonly #chunker: Chunker;

    // Outbound frame queue for when credit is exhausted
    readonly #pendingSends: Array<{ payload: unknown; chunkType: ChunkType }> = [];

    #onFrameOut: ((frame: Frame, transfer?: ArrayBuffer[]) => void) | null = null;
    #onChunk: ((chunk: unknown) => void) | null = null;
    #onError: ((reason: string) => void) | null = null;

    // Sequence number for outbound control frames (not DATA — chunker owns DATA seqs)
    #outSeq: number = 0;
    // finalSeq from received CLOSE frame — needed to detect FINAL_SEQ_DELIVERED
    #remoteFinalSeq: number | null = null;

    constructor(opts: SessionOptions) {
      this.#channelId = opts.channelId;
      this.#streamId = opts.streamId;
      this.#role = opts.role;
      this.#initialCredit = opts.initialCredit ?? 16;

      this.#reorder = new ReorderBuffer(0, {
        maxReorderBuffer: opts.maxReorderBuffer,
      });

      this.#credit = new CreditWindow({
        initialCredit: this.#role === 'initiator' ? 0 : this.#initialCredit,
        highWaterMark: opts.highWaterMark,
        stallTimeoutMs: opts.stallTimeoutMs,
        onStall: () => this.#handleStall(),
        onCreditNeeded: (grant) => this.#sendCreditFrame(grant),
      });

      this.#chunker = new Chunker(0, {
        channelId: this.#channelId,
        streamId: this.#streamId,
        maxChunkSize: opts.maxChunkSize,
      });
    }

    get state(): StreamState { return this.#state; }
    get desiredSize(): number { return this.#credit.desiredSize; }

    onFrameOut(cb: (frame: Frame, transfer?: ArrayBuffer[]) => void): void {
      this.#onFrameOut = cb;
    }
    onChunk(cb: (chunk: unknown) => void): void { this.#onChunk = cb; }
    onError(cb: (reason: string) => void): void { this.#onError = cb; }

    // ... (open, receiveFrame, sendData, close, cancel, reset methods)
  }
  ```

  **receiveFrame(frame):** Main dispatch. Check terminal state first:
  ```typescript
  receiveFrame(frame: Frame): void {
    if (isTerminalState(this.#state)) return;  // FSM Pitfall 3 — drop silently

    switch (frame.type) {
      case 'OPEN':
        // Responder: sends OPEN_ACK
        this.#transition({ type: 'OPEN_RECEIVED' });
        // Reset reorder buffer to start at initSeq from OPEN frame
        // Actually: reorder buffer initSeq is 0 (will receive DATA from seq=0)
        // Send OPEN_ACK:
        const ack: OpenAckFrame = { [FRAME_MARKER]: 1, channelId: ..., streamId: ...,
          seqNum: this.#outSeq++, type: 'OPEN_ACK', initCredit: this.#initialCredit };
        this.#transition({ type: 'OPEN_ACK_SENT', initCredit: this.#initialCredit });
        this.#onFrameOut?.(ack, []);
        break;

      case 'OPEN_ACK':
        // Initiator receives ACK — gain credits
        this.#credit.addSendCredit(frame.initCredit);
        this.#transition({ type: 'OPEN_ACK_RECEIVED', initCredit: frame.initCredit });
        // Drain any pending sends now that we have credit
        this.#drainPendingSends();
        break;

      case 'DATA':
        this.#credit.addRecvConsumed(1);
        const delivered = this.#reorder.insert(frame);
        for (const df of delivered) {
          const reassembled = this.#chunker.reassemble(df);
          if (reassembled !== null) {
            this.#onChunk?.(reassembled);
            this.#credit.notifyRead();  // SESS-03: credit driven by reads, not arrivals
          }
        }
        this.#transition({ type: 'DATA_RECEIVED' });
        // Check if we've delivered finalSeq (if CLOSE already received)
        if (this.#remoteFinalSeq !== null &&
            !seqGT(this.#remoteFinalSeq, this.#reorder.nextExpected - 1 >>> 0)) {
          this.#transition({ type: 'FINAL_SEQ_DELIVERED' });
        }
        break;

      case 'CREDIT':
        this.#credit.addSendCredit(frame.credit);
        this.#drainPendingSends();
        break;

      case 'CLOSE':
        this.#remoteFinalSeq = frame.finalSeq;
        this.#transition({ type: 'CLOSE_RECEIVED' });
        break;

      case 'CANCEL':
        this.#transition({ type: 'CANCEL_RECEIVED', reason: frame.reason });
        this.#onError?.(frame.reason);
        break;

      case 'RESET':
        this.#transition({ type: 'RESET_RECEIVED', reason: frame.reason });
        this.#onError?.(frame.reason);
        break;
    }
  }
  ```

  **sendData():** Credit-gated, queues if blocked:
  ```typescript
  sendData(payload: unknown, chunkType: ChunkType): void {
    if (!this.#credit.consumeSendCredit()) {
      this.#pendingSends.push({ payload, chunkType });
      return;
    }
    this.#emitData(payload, chunkType);
  }

  #emitData(payload: unknown, chunkType: ChunkType): void {
    const results = this.#chunker.split(payload, chunkType as 'BINARY_TRANSFER' | 'STRUCTURED_CLONE');
    for (const { frame, transfer } of results) {
      this.#onFrameOut?.(frame, transfer);
    }
    this.#transition({ type: 'DATA_SENT' });
  }

  #drainPendingSends(): void {
    while (this.#pendingSends.length > 0 && this.#credit.consumeSendCredit()) {
      const { payload, chunkType } = this.#pendingSends.shift()!;
      this.#emitData(payload, chunkType);
    }
  }
  ```

  **close() / cancel() / reset():**
  ```typescript
  close(): void {
    const frame: CloseFrame = { [FRAME_MARKER]: 1, ..., type: 'CLOSE', finalSeq: /* last sent seq */ 0 };
    this.#transition({ type: 'CLOSE_SENT' });
    this.#onFrameOut?.(frame, []);
  }
  cancel(reason: string): void {
    const frame: CancelFrame = { [FRAME_MARKER]: 1, ..., type: 'CANCEL', reason };
    this.#transition({ type: 'CANCEL_SENT', reason });
    this.#onError?.(reason);
    this.#onFrameOut?.(frame, []);
  }
  reset(reason: string): void {
    const frame: ResetFrame = { [FRAME_MARKER]: 1, ..., type: 'RESET', reason };
    this.#transition({ type: 'RESET_SENT', reason });
    this.#onError?.(reason);
    this.#onFrameOut?.(frame, []);
  }
  ```

```typescript
sendFrame(frame: Frame, transfer?: ArrayBuffer[]): void {
  if (frame.type === 'DATA') {
    this.#lastDataSeqOut = frame.seqNum;
    // Track bytes for BINARY_TRANSFER path (exact); STRUCTURED_CLONE is approximate
    if (frame.chunkType === 'BINARY_TRANSFER' && frame.payload instanceof ArrayBuffer) {
      this.#bytesSent += frame.payload.byteLength;
    }
  }
  // Track frame counts (OBS-01)
  this.#frameCountsSent.set(frame.type, (this.#frameCountsSent.get(frame.type) ?? 0) + 1);

  // Emit trace event if enabled (OBS-03)
  if (this.#options.trace) {
    this.#emitter.emit('trace', {
      timestamp: performance.now(),
      direction: 'out',
      frameType: frame.type,
      streamId: frame.streamId,
      seq: frame.seqNum,
      byteLength:
        frame.type === 'DATA' && frame.chunkType === 'BINARY_TRANSFER' &&
        frame.payload instanceof ArrayBuffer
          ? frame.payload.byteLength
          : undefined,
    });
  }

  this.#sendRaw(encode(frame), transfer ?? []);
}
```

```typescript
 * Returns a snapshot of channel and stream metrics (OBS-01).
 * Polling-based — call as needed; not reactive.

stats(): ChannelStats {
  const streams: StreamStats[] = [];

  if (this.#session !== null) {
    // Access session internals via getters added to Session class (or expose via method)
    streams.push({
      streamId: this.#session.streamId,     // need to expose streamId getter on Session

      frameCountsByType: Object.fromEntries(

          .reduce((acc, [k, v]) => {

            return acc;
          }, new Map<string, number>())
      ) as Partial<Record<FrameType, number>>,
      creditWindowAvailable: this.#session.creditWindowAvailable, // new getter on Session

      chunkerChunksReceived: this.#session.chunkerChunksReceived,  // new getter on Session
    });
  }

  return {
    streams,
    aggregate: {
      bytesSent: this.#bytesSent,
```

```typescript
// tests/integration/observability.test.ts
// OBS-01 — stats() snapshot; OBS-02 — typed error events
import { MessageChannel } from 'node:worker_threads';
import { afterEach, describe, expect, it } from 'vitest';
import { createChannel } from '../../src/channel/channel.js';
import type { PostMessageEndpoint } from '../../src/transport/endpoint.js';

function makePair(channelId: string) {
  const { port1, port2 } = new MessageChannel();
  const chA = createChannel(port1 as unknown as PostMessageEndpoint, { channelId });
  const chB = createChannel(port2 as unknown as PostMessageEndpoint, { channelId });
  return { chA, chB, port1, port2 };
}

describe('Channel — stats() (OBS-01)', () => {
  it('frameCountsByType includes OPEN after opening a stream', async () => {
    const { chA, port1, port2 } = makePair('stats-1');

    await chA.capabilityReady;
    chA.openStream();

    // Let frames propagate
    await new Promise<void>(r => setTimeout(r, 30));

    const stats = chA.stats();
    expect(stats.streams).toHaveLength(1);
    expect(stats.streams[0]!.frameCountsByType['OPEN']).toBeGreaterThanOrEqual(1);

    port1.close();
    port2.close();
  });

  it('stats() aggregate bytesSent increases after sending data', async () => {
    const { chA, chB, port1, port2 } = makePair('stats-2');

    await Promise.all([chA.capabilityReady, chB.capabilityReady]);

    const handle = chA.openStream();
    await new Promise<void>(r => setTimeout(r, 20)); // wait for OPEN_ACK

    const before = chA.stats().aggregate.bytesSent;
    const buf = new ArrayBuffer(1024);
    handle.session.sendData(buf, 'BINARY_TRANSFER');

    await new Promise<void>(r => setTimeout(r, 20));

    const after = chA.stats().aggregate.bytesSent;
    expect(after).toBeGreaterThan(before);

    port1.close();
    port2.close();
  });

  it('stats() reorderBufferDepth is 0 after all in-order frames delivered', async () => {
    const { chA, chB, port1, port2 } = makePair('stats-3');

    await Promise.all([chA.capabilityReady, chB.capabilityReady]);

    chA.openStream();
    await new Promise<void>(r => setTimeout(r, 40));

    // All frames delivered in order — buffer should be empty
    const stats = chA.stats();
    if (stats.streams.length > 0) {
      expect(stats.streams[0]!.reorderBufferDepth).toBe(0);
    }

    port1.close();
    port2.close();
  });

  it('stats() creditWindowAvailable is a non-negative number', async () => {
    const { chA, chB, port1, port2 } = makePair('stats-4');
    await Promise.all([chA.capabilityReady, chB.capabilityReady]);

    chA.openStream();
    await new Promise<void>(r => setTimeout(r, 30));

    const stats = chA.stats();
    if (stats.streams.length > 0) {
      expect(stats.streams[0]!.creditWindowAvailable).toBeGreaterThanOrEqual(0);
    }

    port1.close();
    port2.close();
  });
});

describe('Channel — error event routing (OBS-02)', () => {
  it('CREDIT_DEADLOCK surfaces as typed error event on channel', async () => {
    const { port1, port2 } = new MessageChannel();
    // Very short stall timeout to trigger CREDIT_DEADLOCK quickly
    const chA = createChannel(port1 as unknown as PostMessageEndpoint, {
      channelId: 'obs-stall',
      sessionOptions: { initialCredit: 1, stallTimeoutMs: 50 },
    });
    await chA.capabilityReady;

    const errors: string[] = [];
    chA.on('error', (e) => errors.push(e.code));

    const handle = chA.openStream();
    // Wait for OPEN_ACK
    await new Promise<void>(r => setTimeout(r, 30));
    // Exhaust credits
    handle.session.sendData('chunk', 'STRUCTURED_CLONE');
    // Wait past stall timeout
    await new Promise<void>(r => setTimeout(r, 100));

    expect(errors).toContain('CREDIT_DEADLOCK');

    port1.close();
    port2.close();
  });
});
```

```typescript
describe('Channel — trace events (OBS-03)', () => {
  it('emits trace events when trace:true', () => {
    const ep = makeFakeEndpoint(); // existing helper
    const ch = createChannel(ep, { channelId: 'trace-on', trace: true });
    const traces: unknown[] = [];
    ch.on('trace', (t) => traces.push(t));

    // Simulate inbound CAPABILITY message (triggers inbound trace)
    ep.simulateMessage({
      __ibf_v1__: 1, channelId: 'trace-on', streamId: 0, seqNum: 0,
      type: 'CAPABILITY', protocolVersion: 1, sab: false, transferableStreams: false,
    });

    // Should have at least one trace event (outbound CAPABILITY on construction, plus inbound)
    expect(traces.length).toBeGreaterThanOrEqual(1);
    const inboundTrace = traces.find(
      (t: unknown) => (t as { direction: string }).direction === 'in',
    );
    expect(inboundTrace).toBeDefined();
    expect((inboundTrace as { frameType: string }).frameType).toBe('CAPABILITY');
    expect((inboundTrace as { timestamp: number }).timestamp).toBeTypeOf('number');

    ch.close();
  });

  it('does NOT emit trace events when trace option is absent', () => {
    const ep = makeFakeEndpoint();
    const ch = createChannel(ep, { channelId: 'trace-off' }); // no trace option
    const traces: unknown[] = [];
    ch.on('trace', (t) => traces.push(t));

    ep.simulateMessage({
      __ibf_v1__: 1, channelId: 'trace-off', streamId: 0, seqNum: 0,
      type: 'CAPABILITY', protocolVersion: 1, sab: false, transferableStreams: false,
    });

    expect(traces).toHaveLength(0);
    ch.close();
  });
});
```

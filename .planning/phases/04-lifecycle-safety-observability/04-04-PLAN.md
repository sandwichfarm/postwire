it('calls onOriginRejected when message arrives from wrong origin', () => {
  // Create a fake window target
  const fakeTarget = new EventTarget() as EventTarget & {
    postMessage?: () => void;
    dispatchEvent: (e: Event) => boolean;
  };

  // createWindowEndpoint needs a Window — use a fake with addEventListener
  const fakeWin = {
    postMessage: () => {},
    addEventListener: (type: string, fn: EventListenerOrEventListenerObject) =>
      fakeTarget.addEventListener(type, fn),
    removeEventListener: (type: string, fn: EventListenerOrEventListenerObject) =>
      fakeTarget.removeEventListener(type, fn),
  } as unknown as Window;

  const rejected: string[] = [];
  const ep = createWindowEndpoint(fakeWin, 'https://expected.com', {
    onOriginRejected: (origin) => rejected.push(origin),
  });

  // Simulate a message from wrong origin
  const event = Object.assign(new Event('message'), {
    origin: 'https://evil.com',
    data: { __ibf_v1__: 1 },
  }) as MessageEvent;
  fakeTarget.dispatchEvent(event);

  expect(rejected).toEqual(['https://evil.com']);
  expect(ep.onmessage).toBeNull(); // onmessage not called for wrong origin
});

it('REORDER_OVERFLOW caught in receiveFrame routes to onError as REORDER_OVERFLOW', () => {
  const session = new Session({
    channelId: 'reorder-err',
    streamId: 1,
    role: 'responder',
    maxReorderBuffer: 2, // tiny buffer to force overflow quickly
  });
  session.onFrameOut(() => {}); // required

  const errors: string[] = [];
  session.onError((r) => errors.push(r));

  // Open the session first (responder receives OPEN)
  session.receiveFrame({ __ibf_v1__: 1, channelId: 'reorder-err', streamId: 1, seqNum: 0, type: 'OPEN', initCredit: 16 });

  // Inject out-of-order DATA frames to overflow the 2-frame buffer
  // Expected seqNum is 0; inject seq 2, 3, 4 (all out of order, buffer overflows at 3)

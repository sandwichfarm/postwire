```typescript
// Endpoint teardown detection (LIFE-03)
// Node 22: MessagePort fires 'close' when partner closes — detected here.
// Browser: 'close' event on MessagePort is proposal-stage (Blink-only, not cross-browser).
//   In browser, teardown detection for MessagePort falls back to heartbeat (LIFE-02).
//   For Window endpoints, pagehide(persisted=false) covers iframe unload (LIFE-01).
// Adding the listener is always safe — it is a no-op if 'close' never fires.
if (typeof (endpoint as unknown as EventTarget).addEventListener === 'function') {
  const onEndpointClose = (): void => {
    this.#freezeAllStreams('CHANNEL_CLOSED');
  };
  (endpoint as unknown as EventTarget).addEventListener('close', onEndpointClose);
  this.#disposers.push(() =>
    (endpoint as unknown as EventTarget).removeEventListener('close', onEndpointClose),
  );
}

// Push the onmessage teardown as a disposer (LIFE-05 — removes library listener on close)
// We set endpoint.onmessage; removing it means setting it back to null.
this.#disposers.push(() => {
  endpoint.onmessage = null;
});
```

**src/transport/adapters/window.ts — add onOriginRejected callback option:**

```typescript
export interface WindowEndpointOptions {
  /** Called when an inbound message is dropped due to origin mismatch (OBS-02 ORIGIN_REJECTED). */
  onOriginRejected?: (origin: string) => void;
}

export function createWindowEndpoint(
  win: Window,
  expectedOrigin: string,
  opts?: WindowEndpointOptions,
): PostMessageEndpoint {
  // ... existing wildcard and empty-string guards unchanged ...

  const endpoint: PostMessageEndpoint = {
    postMessage(message: unknown, transfer?: Transferable[]): void {
      win.postMessage(message, expectedOrigin, transfer ?? []);
    },
    onmessage: null,
  };

  const listener = (event: MessageEvent): void => {
    if (event.origin !== expectedOrigin) {
      opts?.onOriginRejected?.(event.origin); // Phase 4: surfaces ORIGIN_REJECTED (OBS-02)
      return;
    }
    endpoint.onmessage?.(event);
  };

  win.addEventListener('message', listener);
  // Note: removeEventListener for this listener is the caller's responsibility
  // (Channel wraps it via createWindowEndpoint and adds to #disposers separately).

  return endpoint;
}
```

The Channel that uses createWindowEndpoint passes an onOriginRejected callback that calls `this.#emitter.emit('error', new StreamError('ORIGIN_REJECTED', undefined))`. This wiring is done at the call site (wherever createChannel + createWindowEndpoint are used together), NOT inside createWindowEndpoint itself — the adapter stays unaware of Channel.

```typescript
// tests/integration/lifecycle-teardown.test.ts
// LIFE-03: port close → CHANNEL_CLOSED
// LIFE-04: strong ref retention (implicit — channel holds endpoint ref)
// LIFE-05: listener cleanup after channel.close()
//
// NOTE: The 'close' event on MessagePort is Node-specific.
// Browser teardown detection relies on heartbeat (LIFE-02) and pagehide (LIFE-01).
import { MessageChannel } from 'node:worker_threads';
import { afterEach, describe, expect, it } from 'vitest';
import { createChannel } from '../../src/channel/channel.js';

describe('Channel — endpoint teardown (LIFE-03)', () => {
  const ports: import('node:worker_threads').MessagePort[] = [];
  afterEach(() => {
    // Close any leftover ports to prevent test pollution
    ports.splice(0).forEach(p => { try { p.close(); } catch {} });
  });

  it('emits CHANNEL_CLOSED when the remote port closes', async () => {
    const { port1, port2 } = new MessageChannel();
    ports.push(port1, port2);

    // Cast Node MessagePort to PostMessageEndpoint (same shape)
    const chA = createChannel(port1 as unknown as import('../../src/transport/endpoint.js').PostMessageEndpoint, {
      channelId: 'teardown-1',
    });

    const errors: unknown[] = [];
    chA.on('error', (e) => errors.push(e));

    // Close port2 — simulates remote endpoint dying
    port2.close();

    // Node fires 'close' async (within one event loop tick)
    await new Promise<void>((resolve) => setTimeout(resolve, 20));

    expect(errors).toHaveLength(1);
    expect((errors[0] as { code: string }).code).toBe('CHANNEL_CLOSED');
  });

  it('session is null (no zombie) after port close', async () => {
    const { port1, port2 } = new MessageChannel();
    ports.push(port1, port2);

    const chA = createChannel(port1 as unknown as import('../../src/transport/endpoint.js').PostMessageEndpoint, {
      channelId: 'teardown-2',
    });
    // Open a stream to populate #session
    chA.openStream();
    expect(chA.hasActiveSession).toBe(true); // need to expose this accessor

    port2.close();
    await new Promise<void>((resolve) => setTimeout(resolve, 20));

    // After teardown, no active session remains
    expect(chA.hasActiveSession).toBe(false);
  });

  it('endpoint.onmessage is set to null after channel.close() (LIFE-05)', () => {
    const { port1, port2 } = new MessageChannel();
    ports.push(port1, port2);

    const ep = port1 as unknown as import('../../src/transport/endpoint.js').PostMessageEndpoint;
    createChannel(ep, { channelId: 'teardown-3' });

    // onmessage was set by Channel constructor
    expect(ep.onmessage).not.toBeNull();

    // Closing the channel should clean up the listener
    // We close port1 via its own close method to trigger the channel's close path
    // OR call channel.close() directly
    // channel.close() is called via the port close path tested above;
    // here we explicitly call it:
    // (re-create to avoid contamination from previous test)
    const { port1: p1, port2: p2 } = new MessageChannel();
    ports.push(p1, p2);
    const ep2 = p1 as unknown as import('../../src/transport/endpoint.js').PostMessageEndpoint;
    const ch2 = createChannel(ep2, { channelId: 'teardown-4' });
    expect(ep2.onmessage).not.toBeNull();
    ch2.close();
    expect(ep2.onmessage).toBeNull();
  });
});
```

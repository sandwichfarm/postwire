2. Add #startHeartbeat() method:
```typescript
#startHeartbeat(): void {
  const { intervalMs, timeoutMs } = this.#options.heartbeat!;
  this.#heartbeatInterval = setInterval(() => {
    if (this.#isClosed) return;
    this.#sendCapability(); // ping
    // Arm timeout: if no pong arrives, declare CHANNEL_DEAD
    this.#heartbeatTimeout = setTimeout(() => {
      this.#heartbeatTimeout = null;
      this.#freezeAllStreams('CHANNEL_DEAD');
    }, timeoutMs);
  }, intervalMs);
  // Register cleanup in disposers (LIFE-05)
  this.#disposers.push(() => {
    if (this.#heartbeatInterval !== null) {
      clearInterval(this.#heartbeatInterval);
      this.#heartbeatInterval = null;
    }
    if (this.#heartbeatTimeout !== null) {
      clearTimeout(this.#heartbeatTimeout);
      this.#heartbeatTimeout = null;
    }
  });
}
```

3. Update #handleCapability to handle post-handshake CAPABILITY as ping/pong (RESEARCH.md Pattern 4, Pitfall 3):
```typescript
#handleCapability(frame: CapabilityFrame): void {
  if (frame.protocolVersion !== PROTOCOL_VERSION) {
    const err = new StreamError('PROTOCOL_MISMATCH', undefined);
    this.#rejectCapability(err);
    this.#emitter.emit('error', err);
    this.#onErrorCb?.(err);  // backward compat
    return;
  }
  const isPostHandshake = this.#remoteCap !== null;
  this.#remoteCap = {
    sab: frame.sab && this.#localCap.sab,
    transferableStreams: frame.transferableStreams && this.#localCap.transferableStreams,
  };
  if (!isPostHandshake) {
    this.#resolveCapability();
  } else {
    // Post-handshake CAPABILITY: ping/pong discrimination (RESEARCH.md Pitfall 3)
    // If #heartbeatTimeout is set, we sent this ping and this is the pong — clear timeout.
    // If #heartbeatTimeout is null, the REMOTE sent us a ping — echo back once.
    if (this.#heartbeatTimeout !== null) {
      clearTimeout(this.#heartbeatTimeout);
      this.#heartbeatTimeout = null;
    } else {
      // Remote initiated ping — echo once as pong (do NOT start a timeout)
      this.#sendCapability();
    }
  }
}
```

```typescript
import { afterEach, describe, expect, it, vi } from 'vitest';

describe('Channel — SW heartbeat (LIFE-02)', () => {
  afterEach(() => vi.useRealTimers());

  it('emits CHANNEL_DEAD after timeoutMs when no CAPABILITY pong arrives', () => {
    vi.useFakeTimers();
    const ep = makeFakeEndpoint(); // re-use existing helper from channel.test.ts
    const ch = createChannel(ep, {
      channelId: 'hb-1',
      heartbeat: { intervalMs: 10_000, timeoutMs: 30_000 },
    });
    const errors: unknown[] = [];
    ch.on('error', (e) => errors.push(e));

    vi.advanceTimersByTime(10_001); // trigger first ping interval
    // Verify a CAPABILITY ping was sent (ep.sent[0] = initial CAPABILITY; ep.sent[1] = ping)
    expect(ep.sent.length).toBeGreaterThanOrEqual(2);

    vi.advanceTimersByTime(30_001); // advance past timeout without pong
    expect(errors).toHaveLength(1);
    expect((errors[0] as { code: string }).code).toBe('CHANNEL_DEAD');
  });

  it('does NOT emit CHANNEL_DEAD when CAPABILITY pong arrives before timeout', () => {
    vi.useFakeTimers();
    const ep = makeFakeEndpoint();
    const ch = createChannel(ep, {
      channelId: 'hb-2',
      heartbeat: { intervalMs: 10_000, timeoutMs: 30_000 },
    });
    const errors: unknown[] = [];
    ch.on('error', (e) => errors.push(e));

    vi.advanceTimersByTime(10_001); // ping sent

    // Simulate pong: CAPABILITY frame arriving from remote
    ep.simulateMessage({
      __ibf_v1__: 1,
      channelId: 'hb-2',
      streamId: 0,
      seqNum: 0,
      type: 'CAPABILITY',
      protocolVersion: 1,
      sab: false,
      transferableStreams: false,
    });

    vi.advanceTimersByTime(30_001); // timeout would have fired — but pong reset it
    expect(errors).toHaveLength(0);
    ch.close();
  });

  it('heartbeat timers cleared after channel.close()', () => {
    vi.useFakeTimers();
    const ep = makeFakeEndpoint();
    const ch = createChannel(ep, {
      channelId: 'hb-3',
      heartbeat: { intervalMs: 10_000, timeoutMs: 30_000 },
    });
    const errors: unknown[] = [];
    ch.on('error', (e) => errors.push(e));

    ch.close();
    vi.advanceTimersByTime(200_000); // well past any timeout
    expect(errors).toHaveLength(0);
  });
});
```
